生成新的项目
mvn archetype:create -DgroupId=com.taobao.pamirs -DartifactId=crm-scheme

生成 eclipse项目
mvn eclipse:eclipse

项目生成jar 包 
mvn package -Dtest -DfailIfNoTests=false

项目生成 java doc
mvn package  -Dtest javadoc:javadoc -DfailIfNoTests=false
mvn package  -Dtest -DfailIfNoTests=false

mvn -U -Dtest clean package assembly:assembly  -DfailIfNoTests=false
忽略测试-DfailIfNoTests=false

产生项目原文件
mvn source:jar

生成DocBook
mvn taobaodoc:transform

提交本地库
mvn install

项目的整个依赖树
mvn dependency:tree

mvn eclipse:clean eclipse:eclipse compile package -Dtest -DfailIfNoTests=false

mvn -Dmaven.test.skip clean deploy
